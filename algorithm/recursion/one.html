控制并发请求 ​

看你简历上做过图片或文件批量下载，那么假如我一次性下载几十个，如何去控制并发请求的？

浏览器发起的请求最大并发数量一般都是6~8个，这是因为浏览器会限制同一域名下的并发请求数量，以避免对服务器造成过大的压力。

答案 ​

javascript
/**
 * @param {Promise[]} nums
 * @param {number} concurrency // 最大并发数
*/
const handQueue = (reqs, concurrency = 6) => {
    reqs = reqs || []

    const requestQueue = () => {
        const queue = [] // 请求池
        let current = 0

        const dequeue = () => {
            while (current < concurrency && queue.length) {
                current++;
                console.log('当前并发数', current)
                const requestPromiseFactory = queue.shift() // 出列
                requestPromiseFactory().then((res) => { // 成功的请求逻辑
                    console.log('请求结束-------', res)
                })
                .catch(error => { // 失败
                    console.log(error)
                })
                .finally(() => {
                    current--
                    dequeue()
                });
            }
        }

        return (requestPromiseFactory) => {
            queue.push(requestPromiseFactory) // 入队
            dequeue()
        }
    }

    const enqueue = requestQueue(6)

    // for (let i = 0; i < reqs.length; i++) {
    //     enqueue(() => axios.get('/api/test' + i))
    // }
    for (let i = 0; i < reqs.length; i++) {
        enqueue(reqs[i])
    }
}

// 生成10个promise
function generatePromises(num = 10) {
    const promises = [];
    for (let i = 0; i < num; i++) {
        promises.push(() => {
            return new Promise((resolve) => {
                const delay = Math.floor(Math.random() * 5000); // 随机延迟 0 到 5000 毫秒
                setTimeout(() => {
                    resolve(`Promise ${i + 1} resolved after ${delay} ms`);
                }, delay);
            })
        });
    }
    return promises;
}

// 使用示例
const promises = generatePromises();
handQueue(promises);
