useEffect闭包陷阱 ​

tsx
import { useState,useEffect } from 'react'

function App() {
  const [count, setCount] = useState(0)
  useEffect(() => {
    const timer = setInterval(() => {
      // 执行操作
      setCount(count+1)
    }, 1000);
    
    return () => {
      clearTimeout(timer);
    };
  }, []);
}

这样写的话定时器只执行一次，count最后变成1

tsx
import { useState,useEffect } from 'react'

function App() {
  const [count, setCount] = useState(0)
  const [timers, setTimers] = useState<Array<NodeJS.Timeout>>([]);
  useEffect(() => {
    const timer = setInterval(() => {
      // 执行操作
      setCount(count+1)
    }, 1000);
    timers.push(timer);
    setTimers(timers);
    console.log(timers);
    
    return () => {
      clearInterval(timer);
    };
  }, [count]);
}

这样写的话定时器会执行多次，count会逐渐递增，但是每次count改变都会增加一个定时器，会严重影响浏览器的性能

使用useRef解决闭包陷阱，不会重复生成定时器 ​

tsx
import { useState,useEffect, useRef } from 'react'
import type { MutableRefObject } from 'react'

function App() {
  const [count, setCount] = useState(0)
  const [timers, setTimers] = useState<Array<any>>([]);
  const saveCallBack: MutableRefObject<any> = useRef();

  useEffect(() => {
    function handle() {
      // 执行操作
      setCount(count+1)
    }
    saveCallBack.current = handle
  }, [count])

  useEffect(() => {
    const tick = () => {
      saveCallBack.current();
    };
    const timer = setInterval(tick, 1000);
    timers.push(timer)
    setTimers(timers);
    console.log(timers);
    return () => {
      clearInterval(timer);
    };
  }, [timers]);
  
  

  return (
    <>
      <div className="card">
        count is {count}
      </div>
    </>
  )
}

export default App
